Local Development:
------------------

---------------
# Linux/Mac
export TURSO_DATABASE_URL="libsql://..."
export TURSO_AUTH_TOKEN="..."
./scripts/setup-turso.sh

# Windows PowerShell
$env:TURSO_DATABASE_URL="libsql://..."
$env:TURSO_AUTH_TOKEN="..."
.\scripts\setup-turso.ps1
